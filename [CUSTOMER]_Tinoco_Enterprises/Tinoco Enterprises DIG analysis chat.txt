        rows = list(reader)

    print(f"Total Shipments: {total_rows:,}\n")

    missing_data = defaultdict(int)
    zone_issues = []
    weight_issues = []
    cost_issues = []

    dates = []

    columns = ['ship_date', 'carrier', 'carrier_service', 'destination_postal_code', 'zone', 'weight', 'dimensions']
    
    for idx, row in enumerate(rows, start=2):  # Start at 2 (row 1 is header)
        # Check for missing data
        for col in columns:
            value = row.get(col, '').strip()
            if not value or value == '':
                missing_data[col] += 1
        
        # Extract and validate zone
        zone_val = row.get('zone', '').strip()

                zone_issues.append(f"Row {idx}: Invalid zone '{zone_val}'")
        
        # Extract and validate weight
        weight_val = row.get('weight', '').strip()
        if weight_val:
            # Parse weight (handle "1.5 lb", "0.70 oz", etc.)
            weight_match = re.search(r'([\d.]+)\s*(lb|oz)', weight_val)
            if weight_match:
                weight_num = float(weight_match.group(1))
                unit = weight_match.group(2)
                
                # Convert oz to lbs
                if unit == 'oz':
                    weight_lbs = weight_num / 16
                else:
                    weight_lbs = weight_num
                
                # Flag outliers
                if weight_lbs < 0.1:
                    weight_issues.append(f"Row {idx}: {weight_val} ({weight_lbs:.3f} lbs)")
                elif weight_lbs > 150:
                    weight_issues.append(f"Row {idx}: {weight_val} ({weight_lbs:.1f} lbs)")
        
        # Collect dates for validation
        date_val = row.get('ship_date', '').strip()
        if date_val:
            dates.append(date_val[:10])  # Extract YYYY-MM-DD
    
    # MISSING DATA REPORT
    print("=" * 80)
    print("MISSING DATA ANALYSIS")
    print("=" * 80)

import csv
import re
from collections import defaultdict, Counter
from datetime import datetime

def validate_firstmile_rules(filepath, brand):
    print("\n" + "="*80)
    print(f"FIRSTMILE RULES VALIDATION: {brand.upper()}")
    print("="*80 + "\n")
    
    with open(filepath, 'r') as f:
        reader = csv.DictReader(f)
        rows = list(reader)
    
    total = len(rows)
    print(f"Total Shipments: {total:,}\n")
    
    # Validation counters
    issues = {
        'tracking': [],
        'service': [],
        'zone': [],
        'weight': [],
        'cost': [],
        'date': []
    }
    
    # Service validation
    valid_services = {
        'GroundAdvantage': 'Economy',
        'Ground': 'Economy',
        'Priority': 'Priority',
        'Express': 'Priority',
        'Expedited': 'Expedited',
        '2Day': 'Expedited',
        '3Day': 'Expedited'
    }
    
    service_counts = Counter()
    carrier_counts = Counter()
    zone_dist = Counter()
    weight_dist = {'0-1lb': 0, '1-5lb': 0, '5-10lb': 0, '10-50lb': 0, '50-150lb': 0, '>150lb': 0}
    
    for idx, row in enumerate(rows, start=2):
        # TRACKING - Not in dataset
        # Would check: USPS (9205/9400/9261), FedEx (12-14 digits), UPS (1Z)
        
        # SERVICE validation
        carrier = row.get('carrier', '').strip()
        service = row.get('carrier_service', '').strip()
        carrier_counts[carrier] += 1
        
        mapped_service = None
        for key, value in valid_services.items():
            if key.lower() in service.lower():
                mapped_service = value
                break
        
        service_counts[service] += 1
        
        if not mapped_service:
            issues['service'].append({
                'row': idx,
                'carrier': carrier,
                'service': service,
                'issue': 'Not Economy/Expedited/Priority'
            })
        
        # ZONE validation
        zone_val = row.get('zone', '').strip()
        if not zone_val:
            issues['zone'].append({
                'row': idx,
                'issue': 'Missing zone',
                'carrier': carrier
            })
        else:
            try:
                zone_num = int(zone_val)
                zone_dist[zone_num] += 1
                if zone_num == 0:
                    issues['zone'].append({
                        'row': idx,
                        'zone': zone_num,
                        'issue': 'Zone 0 invalid'
                    })
                elif zone_num > 8:
                    issues['zone'].append({
                        'row': idx,
                        'zone': zone_num,
                        'issue': 'Zone >8 invalid'
                    })
            except ValueError:
                issues['zone'].append({
                    'row': idx,
                    'zone': zone_val,
                    'issue': 'Non-numeric zone'
                })
        
        # WEIGHT validation
        weight_val = row.get('weight', '').strip()
        if weight_val:
            weight_match = re.search(r'([\d.]+)\s*(lb|oz)', weight_val)
            if weight_match:
                weight_num = float(weight_match.group(1))
                unit = weight_match.group(2)
                weight_lbs = weight_num / 16 if unit == 'oz' else weight_num
                
                # Categorize
                if weight_lbs <= 1:
                    weight_dist['0-1lb'] += 1
                elif weight_lbs <= 5:
                    weight_dist['1-5lb'] += 1
                elif weight_lbs <= 10:
                    weight_dist['5-10lb'] += 1
                elif weight_lbs <= 50:
                    weight_dist['10-50lb'] += 1
                elif weight_lbs <= 150:
                    weight_dist['50-150lb'] += 1
                else:
                    weight_dist['>150lb'] += 1
                
                # Flag issues
                if weight_lbs == 0:
                    issues['weight'].append({
                        'row': idx,
                        'weight': weight_val,
                        'issue': '0 lbs - invalid'
                    })
                elif weight_lbs > 150:
                    issues['weight'].append({
                        'row': idx,
                        'weight': weight_val,
                        'lbs': weight_lbs,
                        'issue': 'Freight range (>150 lbs)'
                    })
        
        # COST validation - Not in dataset
        
        # DATE validation - No scan dates in dataset
    
    # PRINT RESULTS
    print("="*80)
    print("1. TRACKING VALIDATION")
    print("="*80)
    print("Status: NOT IN DATASET - Cannot validate tracking numbers")
    print("Impact: Cannot verify carrier format compliance\n")
    
    print("="*80)
    print("2. SERVICE VALIDATION")
    print("="*80)
    
    print("\nCarrier Distribution:")
    for carrier, count in carrier_counts.most_common():
        pct = count * 100.0 / total
        print(f"  {carrier}: {count:,} ({pct:.1f}%)")
    
    print("\nService Mapping to FirstMile Tiers:")
    for service, count in service_counts.most_common():
        pct = count * 100.0 / total
        mapped = None
        for key, value in valid_services.items():
            if key.lower() in service.lower():
                mapped = value
                break
        status = "OK" if mapped else "INVALID"
        tier = mapped if mapped else "UNKNOWN"
        print(f"  {service}: {count:,} ({pct:.1f}%) -> {tier} [{status}]")
    
    if issues['service']:
        print(f"\nFLAGGED SERVICES ({len(issues['service'])} issues):")
        for issue in issues['service'][:5]:
            print(f"  Row {issue['row']}: {issue['carrier']} {issue['service']} - {issue['issue']}")
    else:
        print("\nALL SERVICES VALID")
    
    print("\n" + "="*80)
    print("3. ZONE VALIDATION")
    print("="*80)
    
    print("\nZone Distribution:")
    for zone in sorted(zone_dist.keys()):
        count = zone_dist[zone]
        pct = count * 100.0 / total
        print(f"  Zone {zone}: {count:,} ({pct:.1f}%)")
    
    missing_zones = sum(1 for issue in issues['zone'] if 'Missing' in issue['issue'])
    invalid_zones = len(issues['zone']) - missing_zones
    
    print(f"\nMissing Zones: {missing_zones:,}")
    print(f"Invalid Zones (0 or >8): {invalid_zones}")
    
    if issues['zone']:
        print(f"\nFLAGGED ZONES (first 5 of {len(issues['zone'])}):")
        for issue in issues['zone'][:5]:
            print(f"  Row {issue['row']}: {issue['issue']}")
    else:
        print("\nALL ZONES VALID (1-8)")
    
    print("\n" + "="*80)
    print("4. WEIGHT VALIDATION")
    print("="*80)
    
    print("\nWeight Distribution:")
    for tier, count in weight_dist.items():
        pct = count * 100.0 / total if total > 0 else 0
        print(f"  {tier}: {count:,} ({pct:.1f}%)")
    
    if issues['weight']:
        print(f"\nFLAGGED WEIGHTS ({len(issues['weight'])} issues):")
        for issue in issues['weight'][:10]:
            lbs = issue.get('lbs', 'N/A')
            if isinstance(lbs, float):
                print(f"  Row {issue['row']}: {issue['weight']} ({lbs:.1f} lbs) - {issue['issue']}")
            else:
                print(f"  Row {issue['row']}: {issue['weight']} - {issue['issue']}")
    else:
        print("\nALL WEIGHTS VALID (>0 lbs, <=150 lbs)")
    
    print("\n" + "="*80)
    print("5. COST VALIDATION")
    print("="*80)
    print("Status: NOT IN DATASET - Cannot validate carrier costs")
    print("Impact: Cannot flag underpriced/overpriced shipments\n")

    ship_dates = [row.get('ship_date', '')[:10] for row in rows if row.get('ship_date')]
    if ship_dates:
        print(f"\nShip Date Range:")
        print(f"  First: {min(ship_dates)}")
        print(f"  Last: {max(ship_dates)}")
        print(f"  Unique dates: {len(set(ship_dates))}")
    
    return {
        'total': total,
        'service_issues': len(issues['service']),
        'zone_issues': len(issues['zone']),
        'weight_issues': len(issues['weight']),
        'carrier_counts': dict(carrier_counts),

    }
